CREATE TABLE FIDE_CATEGORIAS_TB (
  V_id_categoria INT GENERATED BY DEFAULT AS IDENTITY,
  V_nombre_categoria VARCHAR2(255),
  CONSTRAINT CATEGORIAS_id_categoria_PK PRIMARY KEY (V_id_categoria)
);

CREATE TABLE FIDE_PRODUCTOS_TB (
  V_id_producto INT GENERATED BY DEFAULT AS IDENTITY,
  V_id_categoria INT,
  V_nombre_producto VARCHAR2(255),
  V_descripcion_producto CLOB,
  V_precio NUMBER(10, 2),
  V_imagen VARCHAR2(255),
  CONSTRAINT PRODUCTOS_id_producto_PK PRIMARY KEY (V_id_producto),
  CONSTRAINT FK_PRODUCTOS_CATEGORIAS FOREIGN KEY (V_id_categoria) REFERENCES FIDE_CATEGORIAS_TB(V_id_categoria)
);
